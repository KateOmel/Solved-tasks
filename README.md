#### Switch it Up!
```javascript
function switchItUp(number){
let result;
switch(number) {
case 0:
  result = "Zero";
  break;
case 1:
  result = "One";
  break;
case 2:
  result = "Two";
  break;
case 3:
  result = "Three";
  break;
case 4:
  result = "Four";
  break;
case 5:
  result = "Five";
  break;
case 6:
  result = "Six";
  break;
case 7:
  result = "Seven";
  break;
case 8:
  result = "Eight";
  break;
case 9:

  break;
}
return result;
}
```

#### Draw stairs
```javascript
function drawStairs(n) {
 let s = '';
 for (let i = 1; i < n; i++) {
  s+= "I\n";
     for (let j = 1; j <= i; j++) {
       s+= ' ';
      }
  }
  s+="I";
  return s;
}
```
